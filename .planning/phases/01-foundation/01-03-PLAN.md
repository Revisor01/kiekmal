     afterAll(async () => {
       await app.close();
     });

         url: "/health",
       });
       expect(response.statusCode).toBe(200);

       expect(body.status).toBe("ok");
       expect(body.timestamp).toBeDefined();
     });
   });

1. `apps/api/Dockerfile` — Multi-stage Build:
   ```dockerfile
   FROM node:20-alpine AS base
   RUN corepack enable && corepack prepare pnpm@9 --activate
   WORKDIR /app

   # Install dependencies
   FROM base AS deps
   COPY package.json pnpm-workspace.yaml pnpm-lock.yaml .npmrc ./
   COPY apps/api/package.json apps/api/package.json
   COPY packages/types/package.json packages/types/package.json
   RUN pnpm install --frozen-lockfile --filter @prayback/api...

   # Build
   FROM base AS build
   COPY --from=deps /app/node_modules ./node_modules
   COPY --from=deps /app/apps/api/node_modules ./apps/api/node_modules
   COPY --from=deps /app/packages/types/node_modules ./packages/types/node_modules
   COPY packages/types/ packages/types/
   COPY apps/api/ apps/api/
   COPY tsconfig.json ./
   RUN cd apps/api && pnpm build

   # Production
   FROM base AS production
   COPY --from=deps /app/node_modules ./node_modules
   COPY --from=deps /app/apps/api/node_modules ./apps/api/node_modules
   COPY --from=deps /app/packages/types ./packages/types
   COPY --from=build /app/apps/api/dist ./apps/api/dist
   COPY apps/api/drizzle ./apps/api/drizzle
   COPY apps/api/package.json ./apps/api/package.json
   COPY package.json pnpm-workspace.yaml ./

   WORKDIR /app/apps/api
   EXPOSE 3000
   CMD ["node", "dist/index.js"]
   ```
   WICHTIG: Der Build-Context ist das REPO-ROOT (nicht apps/api), damit pnpm Workspaces aufgeloest werden.

3. `docker-compose.prod.yml` fuer server.godsapp.de:
   ```yaml
   services:
     api:
       build:
         context: .
         dockerfile: apps/api/Dockerfile
       restart: unless-stopped
       environment:
         DATABASE_URL: postgresql://prayback:${POSTGRES_PASSWORD}@db:5432/prayback
         PORT: "3000"
         NODE_ENV: production
       depends_on:
         db:
           condition: service_healthy
       labels:
         - "traefik.enable=true"
         - "traefik.http.routers.prayback-api.rule=Host(`prayback.godsapp.de`)"
         - "traefik.http.services.prayback-api.loadbalancer.server.port=3000"
       networks:
         - traefik
         - default

     db:
       image: postgis/postgis:16-3.4
       restart: unless-stopped
       environment:
         POSTGRES_USER: prayback
         POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
         POSTGRES_DB: prayback
       volumes:
         - pg_data:/var/lib/postgresql/data
       healthcheck:
         test: ["CMD-SHELL", "pg_isready -U prayback"]
         interval: 5s
         timeout: 5s
         retries: 5

   volumes:
     pg_data:

   networks:
     traefik:
       external: true
   ```
   Das Traefik-Netzwerk `traefik` muss auf dem Server existieren (siehe CLAUDE.md Server-Setup). POSTGRES_PASSWORD wird per .env auf dem Server gesetzt (nicht prayback_dev).
